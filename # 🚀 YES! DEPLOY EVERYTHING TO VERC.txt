### **Step 4: Update Server for Vercel Serverless**

Vercel uses serverless functions. Modify your `server/index.ts`:

**Create a new file: `server/index.ts`** (or update existing):

```typescript
import express, { Request, Response } from 'express';
import cors from 'cors';
import { db } from './db';
import { lessons, users, progress } from '../shared/schema';
import { eq } from 'drizzle-orm';

const app = express();

// Middleware
app.use(cors({
  origin: true, // Vercel handles CORS automatically
  credentials: true
}));

app.use(express.json());

// Health check
app.get('/api/health', (req: Request, res: Response) => {
  res.json({ 
    status: 'ok', 
    timestamp: new Date().toISOString(),
    environment: process.env.NODE_ENV 
  });
});

// Get all lessons
app.get('/api/lessons', async (req: Request, res: Response) => {
  try {
    const allLessons = await db.select().from(lessons);
    res.json(allLessons);
  } catch (error) {
    console.error('Error fetching lessons:', error);
    res.status(500).json({ error: 'Failed to fetch lessons' });
  }
});

// Get lesson by ID
app.get('/api/lessons/:id', async (req: Request, res: Response) => {
  try {
    const lessonId = parseInt(req.params.id);
    const lesson = await db.select()
      .from(lessons)
      .where(eq(lessons.id, lessonId))
      .limit(1);
    
    if (lesson.length === 0) {
      return res.status(404).json({ error: 'Lesson not found' });
    }
    
    res.json(lesson[0]);
  } catch (error) {
    console.error('Error fetching lesson:', error);
    res.status(500).json({ error: 'Failed to fetch lesson' });
  }
});

// Add more routes here...
// Copy all your existing routes from your current server file

// For Vercel serverless
export default app;

// For local development
if (process.env.NODE_ENV !== 'production') {
  const PORT = process.env.PORT || 5000;
  app.listen(PORT, () => {
    console.log(`Server running on http://localhost:${PORT}`);
  });
}
```

### **Step 7: Update Frontend API Configuration**

In your React app, update API calls to use relative URLs:

**Create/Update `client/src/lib/api.ts`:**

```typescript
// For Vercel, use relative URLs (same domain)
const API_BASE_URL = import.meta.env.DEV 
  ? 'http://localhost:5000'  // Local development
  : '';  // Production - same domain

export const api = {
  async get(endpoint: string) {
    const url = `${API_BASE_URL}${endpoint}`;
    console.log('Fetching:', url);
    
    const response = await fetch(url, {
      method: 'GET',
      headers: {
        'Content-Type': 'application/json',
      },
      credentials: 'include',
    });
    
    if (!response.ok) {
      throw new Error(`API Error: ${response.status}`);
    }
    
    return response.json();
  },

  async post(endpoint: string, data: any) {
    const url = `${API_BASE_URL}${endpoint}`;
    
    const response = await fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      credentials: 'include',
      body: JSON.stringify(data),
    });
    
    if (!response.ok) {
      throw new Error(`API Error: ${response.status}`);
    }
    
    return response.json();
  },

  async put(endpoint: string, data: any) {
    const url = `${API_BASE_URL}${endpoint}`;
    
    const response = await fetch(url, {
      method: 'PUT',
      headers: {
        'Content-Type': 'application/json',
      },
      credentials: 'include',
      body: JSON.stringify(data),
    });
    
    if (!response.ok) {
      throw new Error(`API Error: ${response.status}`);
    }
    
    return response.json();
  }
};
```

### **Step 15: Update Database Client**

Update `server/db.ts`:

```typescript
import { drizzle } from 'drizzle-orm/libsql';
import { createClient } from '@libsql/client';

// For local development, use SQLite
// For production, use Turso
const isDev = process.env.NODE_ENV !== 'production';

let db;

if (isDev) {
  // Local SQLite
  const Database = require('better-sqlite3');
  const sqlite = new Database('./preet_english.db');
  db = drizzle(sqlite);
} else {
  // Turso for production
  const client = createClient({
    url: process.env.TURSO_DATABASE_URL!,
    authToken: process.env.TURSO_AUTH_TOKEN!,
  });
  db = drizzle(client);
}

export { db };
```
